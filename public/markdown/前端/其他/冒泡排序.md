/**
 * 冒泡排序
 * 循环判断，如果左边大于右边，就替换位置，这样每一次循环都能找到数组的最大值或最小值
 * 平均时间复杂度为O(n^2)，空间复杂度为O(1)
 * 最快O(n)，最慢O(n^2)，稳定，占用常用内存
 * @param {*} arr 
 * @returns 
 */
function bubbleSort (arr) {
  let len = arr.length
  if (!len) return arr
  for (let i = 0; i < len; i++) {
    for (let j = 0; j < len -i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        // 直接替换，使用结构赋值和数组的特效
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
        // 中间数据
        // const temp = arr[j];
        // arr[j] = arr[j + 1];
        // arr[j + 1] = temp;
      }
    }
    console.log(`第${i+1}次循环`, arr);
  }
  return arr
}

// 单向冒泡，标记在一轮比较汇总中，如果在一轮比较中没有出现需要交互的数据，说明数组已经有序，减少循环次数
function bubbleSort1 (arr) {
  let len = arr.length
  if (!len) return arr
  for (let i = 0; i < len; i++) {
    let mark = true;
    for (let j = 0; j < len -i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        // 直接替换，使用结构赋值和数组的特效
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
        // 中间数据
        // const temp = arr[j];
        // arr[j] = arr[j + 1];
        // arr[j + 1] = temp;
        mark = false;
      }
    }
    console.log(`第${i+1}次循环`, arr);
    if (mark) return arr;
  }
  return arr
}

// 双向冒泡，普通的冒泡排序，在一轮循环中只能找到最大值或者最小值的其中一个，双向冒泡排序则是多一轮的筛选，即找出最大值也找出最小值
function bubbleSort2 (arr) {
  let low = 0;
  let high = arr.length - 1;
  while (low < high) {
    let mark = true;
    // 找到最大值放到右边
    for (let i = low; i < high; i++) {
        if (arr[i] > arr[i + 1]) {
            [arr[i], arr[i + 1]] = [arr[i + 1], arr[i]];
            mark = false;
        }
    }
    high--;
    // 找到最小值放到左边
    for (let j = high; j > low; j--) {
        if (arr[j] < arr[j - 1]) {
            [arr[j], arr[j - 1]] = [arr[j - 1], arr[j]];
            mark = false;
        }
    }
    low++;
    console.log(`第${low}次循环`, arr);
    if (mark) return arr;
  }
}
console.log('普通冒泡', bubbleSort([88,5,7,3,99,4,32,75,12])) // 9次
console.log('单向冒泡', bubbleSort1([88,5,7,3,99,4,32,75,12])) // 5次
console.log('双向冒泡', bubbleSort2([88,5,7,3,99,4,32,75,12])) // 3次
